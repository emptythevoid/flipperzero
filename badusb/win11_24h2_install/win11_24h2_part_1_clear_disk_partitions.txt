REM Pre-stage Windows 24H2 by clearing out all partitions on primary drive (Disk 0)

SHIFT F10
DELAY 1000

REM CTRL-ALT TAB

RIGHT ARROW
DELAY 100
ENTER

REM remove partitions on disk 0
STRING diskpart
ENTER
DELAY 1000

STRING select disk 0
ENTER
DELAY 1000

STRING clean
ENTER
DELAY 5000

REM leave command prompt
ALT F4
DELAY 1000
ALT TAB
DELAY 1000

REM reboot
ALT F4
DELAY 1000
ALT Y
